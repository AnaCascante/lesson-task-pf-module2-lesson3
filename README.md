### Question

```js
var products = [
	{
		name: "Elephant eggs",
		price: 45.0,
	},
	{
		name: "Bucket of water",
		price: 459.99,
	},
	{
		name: "4 cabbages",
		price: 500.95,
	},
];
```

```html
<ul>
	<li>Elephant eggs - 45.00</li>
	<li>Bucket of water - 459.99</li>
	<li>4 cabbages - 500.95</li>
</ul>
```
